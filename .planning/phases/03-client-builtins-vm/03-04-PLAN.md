Create `tests/mcp_builtins.tl` that exercises all MCP builtins:

```
// MCP Builtins Integration Test
// Requires: tl-test-mcp-server binary built

// Get the test server path (passed as argument or use default)
let server_path = if len(args()) > 0 { args()[0] } else { "./target/debug/tl-test-mcp-server" }

// Test 1: Connect to test server
print("Test 1: mcp_connect")
let client = mcp_connect(server_path)
assert(client != nil, "client should not be nil")
print("  PASS: connected")

// Test 2: Server info
print("Test 2: mcp_server_info")
let info = mcp_server_info(client)
assert(info["name"] == "test-server", "server name should be test-server")
assert(info["version"] == "1.0.0", "server version should be 1.0.0")
print("  PASS: server info correct")

// Test 3: Ping
print("Test 3: mcp_ping")
let ping_result = mcp_ping(client)
assert(ping_result == true, "ping should return true")
print("  PASS: ping succeeded")

// Test 4: List tools
print("Test 4: mcp_list_tools")
let tools = mcp_list_tools(client)
assert(len(tools) == 2, "should have 2 tools")
print("  PASS: " + str(len(tools)) + " tools found")

// Test 5: Call echo tool
print("Test 5: mcp_call_tool echo")
let echo_result = mcp_call_tool(client, "echo", {"message": "hello from TL"})
assert(echo_result["is_error"] == false, "echo should not be an error")
print("  PASS: echo returned: " + str(echo_result["content"]))

// Test 6: Call add tool
print("Test 6: mcp_call_tool add")
let add_result = mcp_call_tool(client, "add", {"a": 10, "b": 32})
assert(add_result["is_error"] == false, "add should not be an error")
print("  PASS: add returned: " + str(add_result["content"]))

// Test 7: Disconnect
print("Test 7: mcp_disconnect")
mcp_disconnect(client)
print("  PASS: disconnected")

print("")
print("All MCP builtin tests passed!")
```

Both should produce identical output with all tests passing.
